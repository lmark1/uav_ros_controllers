222 void uav_controller::CascadePID::calculateAttThrustSp(double dt)
223 {
224  // Calculate first row of PID controllers
225  double velocityRefX = _posXPID->compute(
226  getCurrentReference().transforms[0].translation.x, getCurrPosition()[0], dt);
227  double velocityRefY = _posYPID->compute(
228  getCurrentReference().transforms[0].translation.y, getCurrPosition()[1], dt);
229  double velocityRefZ = _posZPID->compute(
230  getCurrentReference().transforms[0].translation.z, getCurrPosition()[2], dt);
231 
232  // Add velocity feed-forward gains
233  velocityRefX += _ffGainVelocityX * getCurrentReference().velocities[0].linear.x;
234  velocityRefY += _ffGainVelocityY * getCurrentReference().velocities[0].linear.y;
235  velocityRefZ += _ffGainVelocityZ * getCurrentReference().velocities[0].linear.z;
236 
237  // Calculate second row of PID controllers
238  double roll = -_velYPID->compute(velocityRefY, getCurrVelocity()[1], dt);
239  double pitch = _velXPID->compute(velocityRefX, getCurrVelocity()[0], dt);
240  double thrust = _velZPID->compute(velocityRefZ, getCurrVelocity()[2], dt);
241  thrust += _hoverThrust;
242 
243  // Add acceleration feed-forward gains
244  roll += _ffGainAccelerationX * getCurrentReference().accelerations[0].linear.x
245  / GRAVITY_ACCELERATION;
246  pitch += _ffGainAccelerationY * getCurrentReference().accelerations[0].linear.y
247  / GRAVITY_ACCELERATION;
248  _yawRef += _ffGainAccelerationZ * getCurrentReference().accelerations[0].linear.z
249  / GRAVITY_ACCELERATION;
250 
251  // Decouple roll and pitch w.r.t. yaw
252  setAttitudeSp(cos(getCurrentYaw()) * roll + sin(getCurrentYaw()) * pitch,
253  cos(getCurrentYaw()) * pitch - sin(getCurrentYaw()) * roll,
254  _yawRef);
255 
256  setThrustSp(thrust);
257 
258  geometry_msgs::Vector3 newMsg;
259  newMsg.x = velocityRefX;
260  newMsg.y = velocityRefY;
261  newMsg.z = velocityRefZ;
262  _velRefPub.publish(newMsg);
263 
264  geometry_msgs::Vector3 vel;
265  vel.x = getCurrVelocity()[0];
266  vel.y = getCurrVelocity()[1];
267  vel.z = getCurrVelocity()[2];
268  _velCurrPub.publish(vel);
269 }
