112 double PID::compute(const double ref_, const double meas_, const double dt_)
113 {
114  double error;
115  double de;
116  double dt;
117 
118  ref = ref_;
119  meas = meas_;
120  dt = dt_;
121 
122  if (firstPass) {
123  // This is the first step of the algorithm
124  // Init time stamp and error
125  error_old = ref - meas;
126  firstPass = false;
127  } else {
128  error = ref - meas;
129  de = error - error_old;// diff error
130  up = kp * error;// proportional term
131 
132  if (ki == 0) {
133  ui = 0;
134  } else {
135  ui = ui_old + ki * error * dt;// integral term
136  }
137 
138  ud = kd * de / dt;// derivative term
139  u = up + ui + ud;
140 
141  if (u > lim_high) {
142  u = lim_high;
143  ui = ui_old;// antiwind up
144  } else if (u < lim_low) {
145  u = lim_low;
146  ui = ui_old;// antiwind up
147  }
148 
149  ui_old = ui;// save ui for next step
150  error_old = error;
151  }
152 
153  return u;
154 }
