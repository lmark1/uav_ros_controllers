378 void uav_reference::CarrotReference::updateCarrotStatus()
379 {
380  // Check if we landed after takeoff happened in automatic takeoff mode
381  if (!_manualTakeoffEnabled && _takeoffHappened && !m_handlerState.getData().armed) {
382  ROS_WARN("CarrotReference::updateCarrotStatus - UAV disarmed, assume LAND happened");
383  _takeoffHappened = false;
384  if (_carrotEnabled) { _carrotOnLand = true; }
385  resetCarrot();
386  }
387 
388  // Detect enable button - rising edge
389  if (getJoyButtons()[_carrotEnabledIndex] == _carrotEnabledValue && !_carrotEnabled) {
390  if (_manualTakeoffEnabled || _takeoffHappened) {
391  _carrotEnabled = true;
392  _carrotOnLand = false;
393  ROS_INFO("CarrotReference::updateCarrotStatus - CARROT_ON_AIR enabled.");
394  resetIntegrators();
395  resetCarrot();
396  } else if (!_carrotOnLand) {
397  ROS_INFO("CarrotReference::updateCarrotStatus - CARROT_ON_LAND enabled.");
398  _carrotEnabled = true;
399  _carrotOnLand = true;
400  resetIntegrators();
401  resetCarrot();
402  }
403  }
404 
405  // Detect enable button - falling edge
406  if (getJoyButtons()[_carrotEnabledIndex] == (1 - _carrotEnabledValue)
407  && _carrotEnabled) {
408  _carrotEnabled = false;
409  _positionHold = false;
410  _carrotOnLand = false;
411  resetIntegrators();
412  ROS_INFO("CarrotRefernce::updateCarrotStatus - carrot disabled.\n");
413  }
414 
415  // Publish carrot status.
416  std_msgs::String status;
417  if (_positionHold && _carrotEnabled) {
418  status.data = POS_HOLD;
419  } else if (!_positionHold && _carrotEnabled) {
420  if (_carrotOnLand) {
421  status.data = CARROT_ON_LAND;
422  } else {
423  status.data = CARROT_ON_AIR;
424  }
425  } else {
426  status.data = CARROT_OFF;
427  }
428 
429  _pubCarrotStatus.publish(status);
430 }
